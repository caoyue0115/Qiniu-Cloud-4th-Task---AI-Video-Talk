## 一、技术选型策略

### 推荐方案：Gradio + Python（最快出Demo）

```
为什么选Gradio？
✅ 一行代码就能生成Web界面，自带摄像头/麦克风支持
✅ 纯Python，无需前端知识
✅ 快速迭代，适合实训时间有限
✅ 支持实时视频流处理

为什么不选其他方案？
❌ 原生App（Flutter/React Native）— 开发周期长，实训时间不够
❌ 纯Web前端（JS）— 需要额外处理后端AI能力
❌ Streamlit — 对摄像头/麦克风支持不如Gradio
```

#### 3.1 视觉处理 (utils/vision.py)

```python
import cv2
import numpy as np
from PIL import Image
import io
import base64

def capture_frame(video_frame):
    """从视频帧中捕获当前帧"""
    return video_frame

def compress_image(image, max_size=720):
    """压缩图片以减少API调用成本"""
    h, w = image.shape[:2]
    if max(h, w) > max_size:
        scale = max_size / max(h, w)
        new_size = (int(w * scale), int(h * scale))
        image = cv2.resize(image, new_size)
    return image

def detect_color(image):
    """端侧颜色识别（HSV）"""
    hsv = cv2.cvtColor(image, cv2.COLOR_RGB2HSV)
    
    # 定义基本颜色范围
    color_ranges = {
        '红色': [(0, 50, 50), (10, 255, 255)],
        '橙色': [(11, 50, 50), (25, 255, 255)],
        '黄色': [(26, 50, 50), (35, 255, 255)],
        '绿色': [(36, 50, 50), (85, 255, 255)],
        '蓝色': [(86, 50, 50), (125, 255, 255)],
        '紫色': [(126, 50, 50), (155, 255, 255)],
        '粉色': [(156, 50, 50), (170, 255, 255)],
        '白色': [(0, 0, 200), (180, 30, 255)],
        '灰色': [(0, 0, 50), (180, 30, 200)],
        '黑色': [(0, 0, 0), (180, 255, 50)],
        '棕色': [(10, 50, 50), (20, 200, 150)],
    }
    
    max_pixels = 0
    detected_color = "未知"
    
    for color_name, (lower, upper) in color_ranges.items():
        mask = cv2.inRange(hsv, np.array(lower), np.array(upper))
        pixels = cv2.countNonZero(mask)
        if pixels > max_pixels:
            max_pixels = pixels
            detected_color = color_name
    
    return detected_color

def frame_to_base64(image):
    """将OpenCV图像转为base64（用于API调用）"""
    _, buffer = cv2.imencode('.jpg', image)
    return base64.b64encode(buffer).decode('utf-8')
```

#### 3.2 AI服务 (utils/ai_service.py)

```python
import openai
from openai import OpenAI
import base64
from io import BytesIO
from PIL import Image
import numpy as np

class AIService:
    def __init__(self, api_key):
        self.client = OpenAI(api_key=api_key)
        self.conversation_history = []
        self.max_history = 6  # 保留最近3轮对话
    
    def transcribe_audio(self, audio_file):
        """语音转文字（Whisper）"""
        with open(audio_file, "rb") as f:
            transcript = self.client.audio.transcriptions.create(
                model="whisper-1",
                file=f,
                language="zh"
            )
        return transcript.text
    
    def analyze_image(self, image, user_text):
        """视觉理解 + 对话（GPT-4o Vision）"""
        
        # 构建消息
        messages = [
            {
                "role": "system",
                "content": """你是一个AI视觉助手，专门帮助视障人士理解他们周围的世界。
请根据用户看到的画面和用户的问题，用中文给出清晰、简洁的回答。

回答要求：
1. 语言简洁明了，适合语音播报
2. 描述准确，不确定时说明"我看到的可能是..."
3. 涉及安全问题时（如障碍物），给出明确提示
4. 识别药品时，说明药品名称和基本用途，但提醒"请遵医嘱"
5. 回答控制在50字以内，方便语音播报"""
            }
        ]
        
        # 添加上下文历史
        for msg in self.conversation_history[-self.max_history:]:
            messages.append(msg)
        
        # 添加当前用户消息（含图片）
        img_base64 = self._image_to_base64(image)
        messages.append({
            "role": "user",
            "content": [
                {"type": "text", "text": user_text},
                {"type": "image_url", "image_url": {
                    "url": f"data:image/jpeg;base64,{img_base64}",
                    "detail": "low"  # low detail = 更低的成本
                }}
            ]
        })
        
        # 调用API
        response = self.client.chat.completions.create(
            model="gpt-4o-mini",  # 先用mini版控制成本
            messages=messages,
            max_tokens=200,
            temperature=0.7
        )
        
        answer = response.choices[0].message.content
        
        # 更新对话历史
        self.conversation_history.append({
            "role": "user",
            "content": user_text
        })
        self.conversation_history.append({
            "role": "assistant",
            "content": answer
        })
        
        return answer
    
    def text_to_speech(self, text):
        """文字转语音（TTS）"""
        response = self.client.audio.speech.create(
            model="tts-1",
            voice="alloy",
            input=text
        )
        return response.content
    
    def _image_to_base64(self, image):
        """将numpy图像转为base64"""
        pil_img = Image.fromarray(image)
        buffer = BytesIO()
        pil_img.save(buffer, format="JPEG", quality=80)
        return base64.b64encode(buffer.getvalue()).decode('utf-8')
    
    def get_cost_estimate(self):
        """估算成本（简化版）"""
        # GPT-4o-mini: $0.15/1M input tokens, $0.6/1M output tokens
        # 粗略估计每次调用约500 tokens
        return 0.15 * 500 / 1000000  # 约$0.000075/次
```

#### 3.3 主程序 (app.py)

```python
import gradio as gr
import numpy as np
import cv2
import os
from dotenv import load_dotenv
from utils.vision import capture_frame, compress_image, detect_color, frame_to_base64
from utils.ai_service import AIService
from utils.cost_tracker import CostTracker

load_dotenv()

# 初始化
ai_service = AIService(api_key=os.getenv("OPENAI_API_KEY"))
cost_tracker = CostTracker()

# 全局变量
current_frame = None
conversation_history_display = []

def process_question(video_frame, audio_file, question_text):
    """处理用户问题"""
    global current_frame, conversation_history_display
    
    # 1. 获取当前帧
    if video_frame is not None:
        current_frame = compress_image(video_frame)
    else:
        return "请先打开摄像头", None, conversation_history_display
    
    # 2. 获取用户输入
    user_text = ""
    if audio_file is not None:
        # 语音转文字
        user_text = ai_service.transcribe_audio(audio_file)
    elif question_text and question_text.strip():
        user_text = question_text
    else:
        return "请说话或输入文字", None, conversation_history_display
    
    # 3. 检测问题类型
    is_color_question = any(kw in user_text for kw in ["颜色", "什么颜色", "啥颜色"])
    
    if is_color_question:
        # 端侧颜色识别
        color = detect_color(current_frame)
        answer = f"我看到的是{color}"
    else:
        # 云端AI理解
        answer = ai_service.analyze_image(current_frame, user_text)
    
    # 4. 语音合成
    audio_response = ai_service.text_to_speech(answer)
    
    # 5. 更新对话历史
    conversation_history_display.append(f"🧑 你：{user_text}")
    conversation_history_display.append(f"🤖 AI：{answer}")
    
    # 6. 追踪成本
    cost_tracker.add_call()
    
    return answer, audio_response, conversation_history_display

def clear_history():
    """清除对话历史"""
    global conversation_history_display
    conversation_history_display = []
    ai_service.conversation_history = []
    return [], "对话已清除"

# Gradio界面
with gr.Blocks(title="AI视觉对话助手", theme=gr.themes.Soft()) as demo:
    gr.Markdown("# 👁️ AI视觉对话助手")
    gr.Markdown("面向视障人士的AI视觉助手 — 用语音与AI对话，让AI帮你'看'世界")
    
    with gr.Row():
        with gr.Column(scale=1):
            # 摄像头
            video = gr.Image(sources=["webcam"], streaming=True, label="摄像头画面")
            
            # 音频输入
            audio = gr.Audio(sources=["microphone"], type="filepath", label="语音输入")
            
            # 文字输入（备选）
            text_input = gr.Textbox(label="文字输入（备选）", placeholder="或者在这里打字提问...")
            
            # 按钮
            with gr.Row():
                submit_btn = gr.Button("🎤 提问", variant="primary")
                clear_btn = gr.Button("🗑️ 清除对话")
        
        with gr.Column(scale=1):
            # 回答显示
            answer_text = gr.Textbox(label="AI回答", lines=4)
            
            # 语音播放
            audio_output = gr.Audio(label="语音回答", type="numpy", autoplay=True)
            
            # 对话历史
            history = gr.Chatbot(label="对话历史", height=400)
            
            # 成本统计
            cost_display = gr.Markdown("**成本统计**：暂无数据")
    
    # 事件绑定
    submit_btn.click(
        fn=process_question,
        inputs=[video, audio, text_input],
        outputs=[answer_text, audio_output, history]
    )
    
    clear_btn.click(
        fn=clear_history,
        inputs=[],
        outputs=[history, answer_text]
    )

if __name__ == "__main__":
    demo.launch(server_name="0.0.0.0", server_port=7860, share=True)
```

### 5.1 端侧YOLO物品检测

```python
# 安装：pip install ultralytics
from ultralytics import YOLO

# 加载预训练模型（轻量版）
model = YOLO('yolov8n.pt')  # nano版本，手机端可运行

def detect_objects(image):
    results = model(image)
    detections = []
    for r in results:
        for box in r.boxes:
            cls = int(box.cls[0])
            conf = float(box.conf[0])
            if conf > 0.5:
                detections.append({
                    'name': model.names[cls],
                    'confidence': conf
                })
    return detections
```

        """生成缓存key（图像hash + 文本）"""

### 5.3 成本追踪

```python
class CostTracker:
    def __init__(self):
        self.total_calls = 0
        self.total_cost = 0.0
    
    def add_call(self, cost=0.001):
        self.total_calls += 1
        self.total_cost += cost
    
    def get_stats(self):
        return {
            "总调用次数": self.total_calls,
            "总成本($)": round(self.total_cost, 4),
            "平均成本($)": round(self.total_cost / max(self.total_calls, 1), 6)
        }
```
